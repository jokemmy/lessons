### 类型判断 typeof

<div>
  {[
    [ 3, 'number' ], [{ toString: function() { return '{}' }}, 'object' ],
    [ '', 'string' ], [ true, 'boolean' ], [ null, 'object' ],
    [ undefined, 'undefined' ], [ Symbol( 'symbol' ), 'symbol' ],
    [ Object.assign([], { toString: function() { return '[]' }}), 'object' ],
    [ Object.assign( function() {}, { toString: function() { return 'function() {}' }}), 'function' ]
  ].map(([ value, type ], i ) => {
    return (
      <Flex key={i} p={1} alignItems="center" width="60vw">
        typeof {String( value )} === '{type}'
        <Box mx='auto' />
        {String( typeof value === type )}
      </Flex>
    );
  })}
</div>
